<script>

const canvas = document.getElementById("canvas")

const ctx = canvas.getContext("2d")


const upload = document.getElementById("upload")

const speedInput = document.getElementById("speed")

const scaleInput = document.getElementById("scale")

const flipInput = document.getElementById("flip")

const slogan = document.getElementById("slogan")


let img = null

let t = 0


const slogans = [

  "🔥 黄大厨开火！",

  "🍳 爆炒进行中...",

  "🥢 翻锅！加点料！",

  "🌶️ 香味出来了！",

  "🍲 出锅啦！"

]


upload.onchange = e => {

  const file = e.target.files[0]

  if (!file) return


  const reader = new FileReader()

  reader.onload = () => {

    img = new Image()

    img.src = reader.result

  }

  reader.readAsDataURL(file)

}


function draw() {

  requestAnimationFrame(draw)

  ctx.clearRect(0,0,256,256)

  if (!img) return


  const speed = parseFloat(speedInput.value)

  const scaleUser = parseFloat(scaleInput.value)

  const flip = flipInput.checked ? -1 : 1


  const cx = 128

  const cy = 128


  const baseScale = Math.min(200/img.width,200/img.height)


  const press = Math.sin(t * 0.2 * speed)

  const scaleY = 1 - press * 0.2

  const scaleX = 1 + press * 0.1


  const yOffset = press * 10


  ctx.save()

  ctx.translate(cx, cy + yOffset)

  ctx.scale(flip * scaleX * baseScale * scaleUser, scaleY * baseScale * scaleUser)


  ctx.drawImage(img, -img.width/2, -img.height/2)


  ctx.restore()


  if (t % 30 === 0) {

    slogan.innerText = slogans[Math.floor(Math.random()*slogans.length)]

  }


  t++

}


draw()


document.getElementById("export").onclick = () => {

  if (!img) return alert("先上传图片")


  const gif = new GIF({

    workers: 2,

    quality: 10,

    workerScript: "https://cdn.jsdelivr.net/npm/gif.js.optimized/dist/gif.worker.js"

  })


  let frames = 50

  let tempT = 0


  for (let i=0;i<frames;i++) {


    ctx.clearRect(0,0,256,256)


    const speed = parseFloat(speedInput.value)

    const scaleUser = parseFloat(scaleInput.value)

    const flip = flipInput.checked ? -1 : 1


    const cx = 128

    const cy = 128


    const baseScale = Math.min(200/img.width,200/img.height)


    const press = Math.sin(tempT * 0.2 * speed)

    const scaleY = 1 - press * 0.2

    const scaleX = 1 + press * 0.1

    const yOffset = press * 10


    ctx.save()

    ctx.translate(cx, cy + yOffset)

    ctx.scale(flip * scaleX * baseScale * scaleUser, scaleY * baseScale * scaleUser)

    ctx.drawImage(img, -img.width/2, -img.height/2)

    ctx.restore()


    gif.addFrame(canvas, {delay: 20})

    tempT++

  }


  gif.on("finished", blob => {

    const url = URL.createObjectURL(blob)


    // 下载

    const a = document.createElement("a")

    a.href = url

    a.download = "huangchef.gif"

    a.click()


    // 提供链接

    const link = document.getElementById("previewLink")

    link.href = url

    link.innerText = "点击打开GIF或右键保存：" + url

  })


  gif.render()

}

</script>
